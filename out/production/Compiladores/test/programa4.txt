##Aritmetica + asignaciones + while##
    print("Bienvenido al programa 4!");
    ulong N := 10UL;
    ulong ACC := 0UL;

        ACC := ACC + N;
        N := N - 1UL;
    };

    }endif;

    print("Suma de 1..10: ");
    print(ACC);
}
